def build_model_simple(classifier):
    assembler = VectorAssembler(inputCols=feature_columns, outputCol="features")
    scaler = MinMaxScaler(inputCol="features", outputCol="scaled_features")
    pipeline = Pipeline(stages=[assembler, scaler, classifier])
    
    return pipeline

classifiers = [
    LogisticRegression(featuresCol="scaled_features", labelCol="isChurn"),
    RandomForestClassifier(featuresCol="scaled_features", labelCol="isChurn"),
    GBTClassifier(featuresCol="scaled_features", labelCol="isChurn")
]

# Lists to store training times and F1 scores
training_times = []
f1_scores = []

# Loop through the classifiers
for classifier in classifiers:
    # Build the model without hyperparameter tuning
    simple_model = build_model_simple(classifier)
    
    # Record the start time
    start_time = time.time()
    
    # Fit the model to the training data
    fitted_model = simple_model.fit(training_data)
    
    # Record the end time
    end_time = time.time()
    
    # Calculate training time
    elapsed_time = end_time - start_time
    training_times.append(elapsed_time)
    
    # Make predictions on the validation data
    predictions = fitted_model.transform(test_data)
    
    # Evaluate the model performance
    evaluator = MulticlassClassificationEvaluator(labelCol='isChurn', metricName='f1')
    f1_score = evaluator.evaluate(predictions)
    f1_scores.append(f1_score)
    
    # Print the training time and F1 score for each classifier
    print(f"{classifier.__class__.__name__}: Training Time = {elapsed_time:.2f} seconds, F1 Score = {f1_score:.4f}")

def build_model_with_tuning(classifier, param_grid):
    assembler = VectorAssembler(inputCols=feature_columns, outputCol="features")
    scaler = MinMaxScaler(inputCol="features", outputCol="scaled_features")
    pipeline = Pipeline(stages=[assembler, scaler, classifier])
    
    cross_validator = CrossValidator(
        estimator=pipeline,
        estimatorParamMaps=param_grid,
        evaluator=MulticlassClassificationEvaluator(labelCol='isChurn', metricName='f1'),
        numFolds=5,
    )
     
    return cross_validator
